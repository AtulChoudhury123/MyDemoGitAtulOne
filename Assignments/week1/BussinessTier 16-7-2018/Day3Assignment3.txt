//Bank Manager class

class BankManager{
	
	static List<Account> l=new List<Account>();
	
	public static bool createAccount(Account acc){
		Console.WriteLine("Enter accNo");
		acc.AccountNo=Convert.ToInt32(Console.ReadLine());
		Console.WriteLine("Enter accType");
		acc.AccountType=Console.ReadLine();
		Console.WriteLine("Enter debitcardno");
		acc.DebitCardNo=Convert.ToInt32(Console.ReadLine());
		Console.WriteLine("Enter pin");
		acc.Pin=Convert.ToInt32(Console.ReadLine());
		Console.WriteLine("Enter loginid");
		acc.LoginId=Console.ReadLine();
		Console.WriteLine("Enter password");
		acc.Password=Console.ReadLine();
		//Console.WriteLine("Enter balance");
		//acc.Balance=Convert.ToInt32(Console.ReadLine());
		
		l.Add(acc);
		
		return true;
	}
	
	public Account getAccountById(int n){
		Account a1=null;
		foreach(Account i in l){
			if(i.AccountNo==n){
				a1=i;
				break;
			}
		}
		
		
		return a1;
	}
	
	public List<Account> getAllAccounts(){
		return l;
	}
}


public class Program
{
	public static void Main()
	{
		Account[] a=new Account[2];
		BankManager bm =new BankManager();
		for(int i=0;i<a.Length;i++){
			a[i]=new Account();
			if(BankManager.createAccount(a[i])){
