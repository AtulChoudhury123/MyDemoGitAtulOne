class AccountMgr{
	public Account createAccount(Account acc){
		
		acc.AccountNo=Convert.ToInt32(Console.ReadLine());
		acc.AccountType=Console.ReadLine();
		acc.DebitCardNo=Convert.ToInt32(Console.ReadLine());
		acc.Pin=Convert.ToInt32(Console.ReadLine());
		acc.LoginId=Console.ReadLine();
		acc.Password=Console.ReadLine();
		acc.Balance=Double.Parse(Console.ReadLine());
		return acc;
	}
	
	public void showData(Account acc){
		Console.WriteLine("account no={0} , accounttype={1}, debitcardno={2}, pin={3}, login={4}, password={5} , balance={6}",
						 acc.AccountNo,acc.AccountType,acc.DebitCardNo,acc.Pin,acc.LoginId,acc.Password,acc.Balance);
	}
}

		Account a2=am.createAccount(a1);
		am.showData(a2);
		
	}
}
