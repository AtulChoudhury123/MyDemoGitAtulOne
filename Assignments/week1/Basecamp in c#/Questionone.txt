using System;
					
public class Program
{
	public static void Main()
	{
		int n=Convert.ToInt32(Console.ReadLine());
		int count=0;
		while(n!=1){
			
		if(n%2==0){
			int x=n/2;
			Console.WriteLine(n+" is even so i take half "+x);
			n=n/2;
		}
		else{
		
			int x=(3*n)+1;
			Console.WriteLine(n+" s odd so i did (3n+1) "+x);
			n=(3*n)+1;
		}
			count++;
		}
		
		Console.WriteLine(count);
		
	}
}
