using System;
	using System.Collections;				
public class Program
{
	public static void Main()
	{
			ArrayList a = new ArrayList();

            bool acceptFlag = true;
            while (acceptFlag)
            {
                Console.WriteLine("Enter some integer");
                int i = int.Parse(Console.ReadLine());
                a.Add(i);// question2
                Console.WriteLine("Do you want to continue? Y/N");
                char accept = char.Parse(Console.ReadLine().ToUpper());
                if (accept == 'Y')
                { acceptFlag = true; }
                else { acceptFlag = false; }
            }
			Console.WriteLine("Number of integers: "+ a.Count);
			int sum=0;
            foreach (int i in a)
            {
				sum=sum+i;
            }
			
		int avg=sum/a.Count;
		int pos=a.Count/2;
		
		a.Insert(pos,avg);
		
		foreach (int i in a)
            {
				Console.WriteLine(i);
            }
		
		a.RemoveAt(pos);
		
		int count=0;
		foreach(int i in a){
			count++;
			if(count==2){
				a.Remove(i);
				break;
			}
		}
		
		foreach (int i in a)
            {
				Console.WriteLine(i);
            }
		
		
	}
}
